<script>
  // Cache for save data string
  window.cachedSaveData = null;

  window.addEventListener("DOMContentLoaded", async () => {
    const ruffle = window.RufflePlayer?.newest();
    if (!ruffle) {
      console.error("Ruffle not found");
      return;
    }

    const container = document.getElementById("flash-container");
    window.player = ruffle.createPlayer();
    container.appendChild(player);

    window.player.style.width = "100%";
    window.player.style.height = "100%";
    window.player.style.backgroundColor = "#000000";
    window.player.load("drbees.swf");

    try {
      window.cachedSaveData = await window.electronAPI.requestSaveData();
    } catch (e) {
      console.error("Failed to cache save data:", e);
      window.cachedSaveData = null;
    }
  });

  // Synchronous function for AS2 ExternalInterface.call to get cached save data
  function loadSaveDataFromElectron() {
    return window.cachedSaveData;
  }

  function unlockSteamAchievement(achName) {
    if (window.steamAPI?.unlockAchievement) {
      window.steamAPI.unlockAchievement(achName);
    }
  }

  window.toggleFullScreen = () => {
    console.log("toggleFullScreen from html!");
    window.electronAPI?.toggleFullscreen?.();
  }

  window.exitGame = () => {
    console.log("exitGame from html!");
    window.electronAPI?.exitGame?.();
  }

  function handleSaveData(jsonString) {
    try {
      const saveObj = JSON.parse(jsonString);
      window.cachedSaveData = jsonString;
      window.electronAPI?.saveGameData(jsonString);
    } catch (e) {
      console.error("Failed to parse save data JSON:", e);
    }
  }

  document.addEventListener("contextmenu", (e) => e.preventDefault());

  // Room data loading bridge for Flash ExternalInterface
  function loadRoomData(roomId) {
    console.log("Loading room:", roomId);
    fetch("./rooms/" + roomId + ".json")
      .then(res => res.json())
      .then(data => {
        const json = JSON.stringify(data);
        console.log("Room data loaded:", json);
        if (window.player && window.player.callExternalInterface) {
          window.player.callExternalInterface("onRoomDataLoaded", json);
        }
      })
      .catch(err => {
        console.error("Failed to load room:", err);
      });
  }
</script>
